系统Dialog

普通Dialog

new AlertDialog.Builder(this)
           .setTitle("title")//设置标题
           .setIcon(R.mipmap.ic_launcher)//设置Icon
           .setMessage("message")//设置正文消息
           .setPositiveButton("positive", new DialogInterface.OnClickListener() {
               @Override
               public void onClick(DialogInterface dialog, int which) {
                   Toast.makeText(getApplicationContext(),"positive",Toast.LENGTH_SHORT).show();
                   dialog.dismiss();
               }
           })
           .setNegativeButton("negative", new DialogInterface.OnClickListener() {
               @Override
               public void onClick(DialogInterface dialog, int which) {
                   Toast.makeText(getApplicationContext(),"negative",Toast.LENGTH_SHORT).show();
                   dialog.dismiss();
               }
           })
           .setNeutralButton("neutral", new DialogInterface.OnClickListener() {
               @Override
               public void onClick(DialogInterface dialog, int which) {
                   Toast.makeText(getApplicationContext(),"neutral",Toast.LENGTH_SHORT).show();
                   dialog.dismiss();
               }
           })
           /*
           这里的which
           Dialog.BUTTON_POSITIVE
           Dialog.BUTTON_NEGATIVE
           Dialog.BUTTON_NEUTRAL
            */
           .create()//创建一个AlertDialog对象
           .show();//显示

列表Dialog

 final String[] items = {"张三", "李四", "王五"};
    new AlertDialog.Builder(this)
            .setTitle("title")//设置标题
            .setIcon(R.mipmap.ic_launcher)//设置Icon
//          .setMessage("message")//设置了message,就不会显示items
            .setItems(items, new DialogInterface.OnClickListener() {
                @Override

多选Dialog

    final String[] items = {"足球","篮球","排球"};
    final boolean[] selected={false,false,true};//当选中或取消选中时,将自动改变selected里对应的值
    new AlertDialog.Builder(this)
            .setTitle("title")//设置标题
            .setIcon(R.mipmap.ic_launcher)//设置Icon
//          .setMessage("message")//设置正文消息,就不会显示items
           .setMultiChoiceItems(items, selected, new DialogInterface.OnMultiChoiceClickListener() {
               @Override
               public void onClick(DialogInterface dialog, int which, boolean isChecked) {
                   Toast.makeText(getApplicationContext(),items[which]+":"+selected[which] , Toast.LENGTH_SHORT).show();
               }
           })
            .setPositiveButton("positive", new DialogInterface.OnClickListener() {
                @Override
                public void onClick(DialogInterface dialog, int which) {
                    Toast.makeText(getApplicationContext(), "positive", Toast.LENGTH_SHORT).show();
                    dialog.dismiss();
                }
            })
            .setNegativeButton("negative", new DialogInterface.OnClickListener() {
                @Override
                public void onClick(DialogInterface dialog, int which) {
                    Toast.makeText(getApplicationContext(), "negative", Toast.LENGTH_SHORT).show();
                    dialog.dismiss();
                }
            })
            .setNeutralButton("neutral", new DialogInterface.OnClickListener() {
                @Override
                public void onClick(DialogInterface dialog, int which) {
                    Toast.makeText(getApplicationContext(), "neutral", Toast.LENGTH_SHORT).show();
                    dialog.dismiss();
                }
            })
            /*
            这里的which
            Dialog.BUTTON_POSITIVE
            Dialog.BUTTON_NEGATIVE
            Dialog.BUTTON_NEUTRAL
             */
            .create()//创建一个AlertDialog对象
            .show();//显示
